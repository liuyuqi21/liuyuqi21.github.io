5. 替换空格

牛客

public String replaceSpace(StringBuffer str) {
    int blank = 0;
    for (int i = 0; i < str.length(); i++) {
        if (str.charAt(i) == ' ')
            blank++;
    }
    int len = str.length();
    int newLen = len + blank * 2;
    str.setLength(newLen);
    newLen--;
    len--;
    while (len >= 0 && newLen > len) {
        char c = str.charAt(len--);
        if (c == ' ') {
            str.setCharAt(newLen--, '0');
            str.setCharAt(newLen--, '2');
            str.setCharAt(newLen--, '%');
        } else {
            str.setCharAt(newLen--, c);
        }
    }
    return str.toString();
}

46. 把数字翻译成字符串

leetcide 91

public int numDecodings(String s) {
        if (s == null || s.length() == 0) {
            return 0;
        }
        int len = s.length();
        int[] dp = new int[len + 1];
        dp[len] = 1;
        if (s.charAt(len - 1) == '0') {
            dp[len - 1] = 0;
        } else {
            dp[len - 1] = 1;
        }
        for (int i = len - 2; i >= 0; i--) {
            if (s.charAt(i) == '0') {
                dp[i] = 0;
                continue;
            }
            if ((s.charAt(i) - '0') * 10 + (s.charAt(i + 1) - '0') <= 26) {
                dp[i] = dp[i + 1] + dp[i + 2];
            } else {
                dp[i] = dp[i + 1];
            }
        }
        return dp[0];
    }

48. 最长不含重复字符的子字符串

leetcode 3

public int lengthOfLongestSubstring(String s) {
    int count = 0;
    int i = 0, j = 0, res = 0;
    Set<Character> set = new HashSet<>();
    while (i < s.length() && j < s.length()) {
        if (!set.contains(s.charAt(j))) {
            set.add(s.charAt(j++));
            res = Math.max(res, j - i);
        } else {
            set.remove(s.charAt(i++));
        }
    }
    return res;
}

50. 第一个只出现一次的字符位置

leetcode 387

public int firstUniqChar(String s) {
    if (s == null || s.length() == 0) {
        return -1;
    }
    if (s.length() == 1) {
        return 0;
    }
    int[] c = new int[256];
    for (int i = 0; i < s.length(); i++) {
        c[s.charAt(i)]++;
    }
    for (int i = 0; i < s.length(); i++) {
        if (c[s.charAt(i)] == 1) {
            return i;
        }
    }
    return -1;
}

58.1 翻转单词顺序列

leetcode 151

public String reverseWords(String s) {
    if (s.length() < 2) {
        return s;
    }
    s = s.trim();
    int length = s.length();
    char[] c = s.toCharArray();
    int start = 0, end = 0;
    reverse(c, start, length - 1);
    while (end <= length) {
        if (end == length || c[end] == ' ') {
            reverse(c, start, end - 1);
            start = end + 1;
        }
        end++;
    }
    return new String(c);
}
public void reverse(char[] c, int start, int end) {
    while (start < end) {
        char temp = c[start];
        c[start] = c[end];
        c[end] = temp;
        start++;
        end--;
    }
}

58.2 左旋转字符串

牛客

public String LeftRotateString(String str, int n) {
    if (str == null || str.length() == 0 || n <= 0) {
        return str;
    }
    char[] c = str.toCharArray();
    reverse(c, 0, n - 1);
    reverse(c, n, str.length() - 1);
    //只能最后翻转整体
    reverse(c, 0, str.length() - 1);
    return new String(c);
}
public void reverse(char[] c, int start, int end) {
    while (start < end) {
        char temp = c[start];
        c[start] = c[end];
        c[end] = temp;
        start++;
        end--;
    }
}
